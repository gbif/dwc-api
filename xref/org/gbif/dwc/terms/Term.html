1   package org.gbif.dwc.terms;
2   
3   import org.codehaus.jackson.map.annotate.JsonDeserialize;
4   import org.codehaus.jackson.map.annotate.JsonSerialize;
5   import org.gbif.dwc.terms.jackson.TermDeserializer;
6   import org.gbif.dwc.terms.jackson.TermSerializer;
7   
8   import java.io.Serializable;
9   import java.net.URI;
10  
11  @JsonSerialize(using= TermSerializer.class)
12  @JsonDeserialize(using= TermDeserializer.class)
13  public interface Term extends Serializable {
14  
15    /**
16     * The simple term name without any namespace or paths.
17     * For example scientificName.
18     */
19    String simpleName();
20  
21    /**
22     * The simple term name prefixed by a short unique namespace abbreviation.
23     * For example dwc:scientificName.
24     */
25    String prefixedName();
26  
27    /**
28     * The full qualified term uri including the namespace.
29     * For example http://rs.tdwg.org/dwc/terms/scientificName.
30     */
31    String qualifiedName();
32  
33    /**
34     * Informs if a term is generally used as a class term, i.e. defining rowTypes not properties.
35     * For example VernacularName, Taxon or Occurrence
36     * @return true if the term is defining a class instead of a property, e.g. Taxon
37     */
38    boolean isClass();
39  
40    /**
41     * A unique standard prefix representing the namespace.
42     * For example dwc.
43     */
44    String prefix();
45  
46    /**
47     * The namespace the terms are in
48     */
49    URI namespace();
50  }
