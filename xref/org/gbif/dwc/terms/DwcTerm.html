279   private DwcTerm(String groupName, String... alternatives) {
280     normQName = TermFactory.normaliseTerm(qualifiedName());
281     for (int i = 0; i < alternatives.length; i++) {
282       alternatives[i] = TermFactory.normaliseTerm(alternatives[i]);
283     }
284     normAlts = alternatives;
285     this.groupName = groupName;
286   }

330   /**
331    * @return true if the dwc term is defining a class instead of a property, e.g. Taxon

333   public boolean isClass() {
334     return Character.isUpperCase(simpleName().charAt(0));
335   }

337   /**
338    * List all terms that belong to a given group.
339    *
340    * @param group the group to list terms for
341    *
342    * @return the list of dwc terms in the given group
343    */
344   public static List<DwcTerm> listByGroup(String group) {
345     List<DwcTerm> terms = new ArrayList<DwcTerm>();
346     for (DwcTerm t : DwcTerm.values()) {
347       if (t.getGroup().equalsIgnoreCase(group)) {
348         terms.add(t);
349       }
350     }
351     return terms;
352   }
