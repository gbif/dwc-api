3   import org.gbif.dwc.terms.Term;
4   import org.gbif.dwc.terms.TermFactory;
5   
6   import java.io.IOException;
7   
8   import org.codehaus.jackson.JsonParser;
9   import org.codehaus.jackson.JsonProcessingException;
10  import org.codehaus.jackson.JsonToken;
11  import org.codehaus.jackson.map.DeserializationContext;
12  import org.codehaus.jackson.map.JsonDeserializer;
13  
14  /**
15   * A json deserializer that turns full qualified term names into dwc terms instances.
16   */
17  public class TermDeserializer extends JsonDeserializer<Term> {
18    private TermFactory factory = TermFactory.instance();
19  
20    @Override
21    public Term deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException, JsonProcessingException {
22      if (jp.getCurrentToken() == JsonToken.VALUE_STRING) {
23        return factory.findTerm(jp.getText());
24      }
25      throw ctxt.mappingException("Expected JSON String");
26    }
27  }
