9   /**
10   * Simple, threadsafe factory for terms that knows about all ConceptTerms of this library and keeps singletons for
11   * all unknown Term instances.
12   */
13  public class TermFactory {
14  
15    private static final Logger LOG = LoggerFactory.getLogger(TermFactory.class);
16    private static final Pattern NON_ALPHA_NUM_PATTERN = Pattern.compile("[^a-zA-Z0-9#-]+");
17    private static TermFactory singleton;
18    private static boolean initialized = false;
19    private static final Object LOCK = new Object();
20  
21    private final Map<String, Term> terms = new HashMap<String, Term>();
22    private final Map<String, Term> classTerms = new HashMap<String, Term>();
23    private final Set<Class<? extends Enum>> registeredEnumClasses = new HashSet<>();
24  
25    public static TermFactory instance() {
26      if (initialized) {
27        return singleton;
28      }
29  
30      synchronized (LOCK) {
31        if (singleton == null) {
32          LOG.debug("Building new TermFactory instance");
33          singleton = new TermFactory();
34          singleton.loadKnownTerms();
35          initialized = true;
36        }
37      }
38  
39      return singleton;
40    }

42    private TermFactory() {
43    }

45    private void loadKnownTerms() {
46      registerTermEnum(DwcTerm.class);
47      registerTermEnum(DcTerm.class, "dct");
48      registerTermEnum(GbifTerm.class);
49      registerTermEnum(GbifInternalTerm.class);
50      registerTermEnum(IucnTerm.class);
51      registerTermEnum(DcElement.class);
52      registerTermEnum(AcefTerm.class);
53      registerTermEnum(AcTerm.class);
54      registerTermEnum(XmpTerm.class, "adobe");
55      registerTermEnum(XmpRightsTerm.class, "xmp", "adobe"); // the same as above, but luckily different simple term names
56    
57      registerQualifiedTermEnum(DwcaTerm.class);
58    }

60    /**
61     * @return the set of term enum classes that have been registered with this TermFactory
62     */
63    public Set<Class<? extends Enum>> listRegisteredTermEnums() {
64      return Collections.unmodifiableSet(registeredEnumClasses);
65    }

75    /**
76     * Registers all terms from a term enumeration.
77     * If the same class is registered again it will be silently ignored.
78     *
79     * @param altPrefixes alternative prefixes to be used to register simple prefixed term names
80     */
81    public <T extends Enum & Term & AlternativeNames> void registerTermEnum(Class<T> termClass, String ... altPrefixes) {
82      if (registeredEnumClasses.contains(termClass)) {
83        LOG.debug("{} is already registered", termClass);
84      } else {
85        registeredEnumClasses.add(termClass);
86        for (T term : termClass.getEnumConstants()) {
87          // add regular term representations (simple, prefixed & qualified)
88          addTerm(term, altPrefixes);
89          // add alternatives
90          for (String alt : term.alternativeNames()) {
91            addTerm(alt, term);
92            addTerm(term.prefix() + ":" + alt, term);
93            addTerm(term.namespace().resolve(alt).toString(), term);
94            for (String pre : altPrefixes) {
95              addTerm(pre + ":" + alt, term);
96            }
97          }
98        }
99      }
100   }

102   /**
103    * Registers all terms from a new term enumeration, but only adds their qualified and prefixed names.
104    * This is to avoid clashes with other usually more important terms that should be known by their simple name.
105    */
106   public <T extends Enum & Term> void registerQualifiedTermEnum(Class<T> termClass) {
107     if (registeredEnumClasses.contains(termClass)) {
108       LOG.debug("{} is already registered", termClass);
109     } else {
110       registeredEnumClasses.add(termClass);
111       for (T term : termClass.getEnumConstants()) {
112         // add only the prefixed and qualified representation to avoid clashes
113         addTerm(term.prefixedName(), term);
114         addTerm(term.qualifiedName(), term);
115       }
116     }
117   }
118   
119   private void addTerm(Term term, String ... altPrefixes) {
120     addTerm(term.simpleName(), term);
121     addTerm(term.prefixedName(), term);
122     addTerm(term.qualifiedName(), term);
123     for (String pre : altPrefixes) {
124       addTerm(pre + ":" + term.simpleName(), term);
125     }
126   }
127 
128   private void addTerm(String key, Term term) {
129     if (key == null || key.trim().isEmpty()) {
130       return;
131     }
132 
133     // keep class terms distinct
134     Map<String, Term> map = termMap(term.isClass());
135     if (map.containsKey(key)) {
136       Term t1 = map.get(key);
137       if (!t1.equals(term)) {
138         LOG.info("{} terms {} and {} are both known as \"{}\". Keeping only earlier {}", term.isClass() ? "Class" : "Property", map.get(key), term, key, map.get(key));
139       }
140     } else {
141       map.put(key, term);
142       // also add a normalised version
143       key = normaliseTerm(key);
144       if (!map.containsKey(key)) {
145         map.put(key, term);
146       }
147     }
148   }
149 
150   private Map<String, Term> termMap(boolean isClass) {
151     return isClass ? classTerms : terms;
152   }
153 
154   /**
155    * @return a purely alphanumerical, lower cased term with all other characters replaced
156    */
157   public static String normaliseTerm(String term) {
158     String x = NON_ALPHA_NUM_PATTERN.matcher(term).replaceAll("");
159     // remove http(s)
160     x = x.replaceFirst("^https?", "");
161     if (x.isEmpty()) {
162       return "";
163     }
164     return x.toLowerCase();
165   }

167   /**
168    * This is the main method to get a term from the factory searching both for property or class terms.
169    * It will lookup matching terms applying some normalization and known synonyms first.
170    * In case of ambiguous terms Class terms will be preferred.
171    *
172    * If nothing matches the factory creates a new UnknownTerm property instance and keeps it for further requests so that
173    * all terms with the same qualified name return a single UnknownTerm instance.
174    *
175    * For clearly bad term names an IllegalArgumentException is thrown.
176    * For example in the case of a simple name containing whitespace like "hello tom".
177    * Ideally the term names to be looked up should be full URIs, but simple names made up of alphanumerics and dashes
178    * will also work fine. Unknown simple names will be put into the namespace http://unknown.org when a new UnknownTerm
179    * instance is created.
180    */
181   public Term findTerm(final String termName) throws IllegalArgumentException {
182     // First try an exact match.
183     if (terms.containsKey(termName)) {
184       return terms.get(termName);
185     }
186 
187     // Try class term
188     Term t = findTermOnly(termName, true);
189     if (t == null) {
190       // Try property term
191       t = findTermOnly(termName, false);
192     }
193     // create new term if needed
194     if (t == null) {
195       t = createUnknownTerm(termName, false);
196     }
197     return t;
198   }
199 
200   /**
201    * This method works just as findTerm(final String termName) but restricts
202    * the results to just property terms.
203    */
204   public Term findPropertyTerm(final String termName) throws IllegalArgumentException {
205     return findTerm(termName, false);
206   }
207 
208   /**
209    * This method works just as findTerm(final String termName) but restricts
210    * the results to just class terms.
211    */
212   public Term findClassTerm(final String termName) throws IllegalArgumentException {
213     return findTerm(termName, true);
214   }
215 
216   /**
217    * This method works just as findTerm(final String termName) but restricts
218    * the results to just property or class terms.
219    */
220   public Term findTerm(final String termName, boolean isClassTerm) throws IllegalArgumentException {
221     if (termName == null || termName.trim().isEmpty()) {
222       return null;
223     }
224 
225     Term t = findTermOnly(termName, isClassTerm);
226     // create new term if needed
227     if (t == null) {
228       t = createUnknownTerm(termName, isClassTerm);
229     }
230     return t;
231   }
232 
233   /**
234    * Does not create Unknown terms
235    */
236   private Term findTermOnly(final String termName, boolean isClassTerm) throws IllegalArgumentException {
237     if (termName == null || termName.trim().isEmpty()) {
238       return null;
239     }
240 
241     Map<String, Term> map = termMap(isClassTerm);
242     // first try term just as it is
243     if (map.containsKey(termName)) {
244       return map.get(termName);
245     }
246 
247     // try normalised term otherwise
248     if (map.containsKey(normaliseTerm(termName))) {
249       return map.get(normaliseTerm(termName));
250     }
251     return null;
252   }
253 
254   private Term createUnknownTerm(String termName, boolean isClassTerm) {
255     // create new term instance
256     Term term = UnknownTerm.build(termName, isClassTerm);
257     addTerm(termName, term);
258     addTerm(term.qualifiedName(), term);
259     return term;
260   }
