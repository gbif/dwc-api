10  /**
11   * Simple factory for terms that knows about all ConceptTerms of this library and keeps singletons for
12   * all unknown Term instances.
13   */
14  public class TermFactory {
15      private final static Logger LOG = LoggerFactory.getLogger(TermFactory.class);
16      private static final Pattern NON_ALPHA_NUM_PATTERN = Pattern.compile("[^a-zA-Z0-9]+");
17      private final Map<String, Term> terms = new HashMap<String, Term>();
18  
19      public TermFactory() {
20          loadKnownTerms();
21      }
22  
23      private void loadKnownTerms() {
24          addTerms(DwcTerm.values(), DwcTerm.PREFIXES);
25          addTerms(DcTerm.values(), DcTerm.PREFIXES);
26          addTerms(GbifTerm.values(), GbifTerm.PREFIXES);
27          addTerms(IucnTerm.values(), IucnTerm.PREFIXES);
28      }
29  
30      private void addTerms(Term[] terms, String[] prefixes) {
31          for (Term term : terms) {
32              addTerm(term.simpleName(), term);
33              addTerm(term.qualifiedName(), term);
34              for (String pre : prefixes) {
35                  addTerm(pre + term.simpleName(), term);
36              }
37              // alt names
38              for (String alt : term.alternativeNames()) {
39                  addTerm(alt, term);
40                  for (String pre : prefixes) {
41                      addTerm(pre + alt, term);
42                  }
43              }
44  
45          }
46      }
47  
48      private void addTerm(String key, Term term) {
49          if (key == null || key.trim().isEmpty()) {
50              return;
51          }
52          key = normaliseTerm(key);
53          if (terms.containsKey(key)) {
54              Term t1 = terms.get(key);
55              if (!t1.equals(term)) {
56                  LOG.warn("Terms {} and {} are both known as \"{}\". Keeping only {}", terms.get(key), term, key, terms.get(key));
57              }
58          } else {
59              terms.put(key, term);
60          }
61      }
62  
63      /**
64       * @return a purely alphanumerical, lower cased term with all other characters replaced
65       */
66      public static String normaliseTerm(String term) {
67          return NON_ALPHA_NUM_PATTERN.matcher(term).replaceAll("").toLowerCase();
68      }
69  
70      public Term findTerm(String termName) {
71          if (termName == null || termName.trim().isEmpty()) {
72              return null;
73          }
74          // normalise terms
75          Term term;
76          String normTermName = normaliseTerm(termName);
77          // try known terms first
78          if (terms.containsKey(normTermName)) {
79              return terms.get(normTermName);
80          } else {
81              // create new term instance
82              int pos = termName.lastIndexOf("/");
83              String tmpName = pos == -1 ? "" : termName.substring(pos + 1);
84              term = new UnknownTerm(termName, tmpName);
85              addTerm(normTermName, term);
86              addTerm(termName, term);
87              addTerm(tmpName, term);
88          }
89          return term;
90      }
