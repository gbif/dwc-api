3   /**
4    * Terms for Adobe XMP have URIs that are not resolvable. Instead, visit
5    * <a href="http://www.adobe.com/content/dam/Adobe/en/devnet/xmp/pdfs/XMPSpecificationPart1.pdf">XMP Specification Part 1, Sec 8.4 </a>
6    * for further documentation.
7    *
8    * These terms are used by the Audubon Core standard.
9    */
10  public enum XmpTerm implements Term, AlternativeNames {
11  
12    MetadataDate,
13    Rating,
14    CreateDate;
15  
16    public static final String NS = "http://ns.adobe.com/xap/1.0/";
17    public static final String PREFIX = "xmp";
18    static final String[] PREFIXES = {NS, PREFIX + ":", "adobe:"};
19  
20    public String toString() {
21      return PREFIX + ":" + this.name();
22    }
23  
24    @Override
25    public String simpleName() {
26      return name();
27    }
28  
29    @Override
30    public String[] alternativeNames() {
31      return new String[]{};
32    }
33  
34    @Override
35    public String qualifiedName() {
36      return NS + simpleName();
37    }
38  
39  }
