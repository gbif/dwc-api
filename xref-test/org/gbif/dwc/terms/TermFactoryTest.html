17    /**
18     * GBIF code assumes a term coming from any of the Term enumerations here have unique simple names.
19     * This tests verifies that!
20     */
21    @Test
22    public void testKnownTermUniqueness() {
23      Set<String> names = new HashSet<String>();
24  
25      addTerms(names, DwcTerm.values());
26      addTerms(names, DcTerm.values());
27      addTerms(names, GbifTerm.values());
28      addTerms(names, GbifInternalTerm.values());
29      addTerms(names, IucnTerm.values());
30  
31      addTerms(names, AcTerm.values());
32      addTerms(names, XmpTerm.values());
33      addTerms(names, XmpRightsTerm.values());
34    }
35  
36    private void addTerms(Set<String> names, Term[] terms) {
37      for (Term t : terms) {
38        assertFalse("Duplicate simple name "+t.simpleName(), names.contains(t.simpleName()));
39        names.add(t.simpleName());
40      }
41    }

43    @Test
44    public void testFindTerm() {
45      TermFactory factory = TermFactory.instance();
46  
47      assertEquals(DwcTerm.scientificName, factory.findTerm("ScientificName"));
48      assertEquals(DwcTerm.scientificName, factory.findTerm("dwc:scientificName"));
49      assertEquals(DwcTerm.scientificName, factory.findTerm("http://rs.tdwg.org/dwc/terms/scientificName"));
50      assertEquals(DcElement.identifier, factory.findTerm("dc:identifier"));
51      assertEquals(DcTerm.identifier, factory.findTerm("identifier"));
52      assertEquals(DcTerm.identifier, factory.findTerm("ID"));
53      assertEquals(DwcTerm.parentNameUsageID, factory.findTerm("dwc:higherTaxonID"));
54  
55      assertEquals(GbifInternalTerm.unitQualifier, factory.findTerm("UNIT_QUALIFIER"));
56  
57      assertEquals("threatStatus", factory.findTerm("http://rs.gbif.org/terms/1.0/threatStatus").simpleName());
58      assertEquals("threatStatus", factory.findTerm("http://rs.gbif.org/terms/1321.43/threatStatus").simpleName());
59  
60      assertEquals(DwcTerm.catalogNumber, factory.findTerm("\"catalogNumber\""));
61    }

114   /**
115      * Not a real test, just a way of running many concurrent TermFactory.instance() calls to verify thread safety.
116      */
117   @Test
118   public void testMultithreadStart() throws InterruptedException {
119     int threadCount = 100;
120     ExecutorService tp = Executors.newFixedThreadPool(threadCount);
121     for (int i = 0; i < threadCount; i++) {
122       tp.submit(new TermFactoryLoader());
123     }
124     tp.shutdown();
125     tp.awaitTermination(30, TimeUnit.SECONDS);
126   }
127 
128   private static class TermFactoryLoader implements Runnable {
129     @Override
130     public void run() {
131       TermFactory.instance();
132     }
133   }
