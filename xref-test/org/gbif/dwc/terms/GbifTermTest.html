44      List<GbifTerm> datasetTerms = GbifTerm.listByGroup(GbifTerm.GROUP_DATASET);

48      List<GbifTerm> occurrenceTerms = GbifTerm.listByGroup(DwcTerm.GROUP_OCCURRENCE);

50      assertEquals(12, new HashSet<GbifTerm>(occurrenceTerms).size());
51  
52      List<GbifTerm> locationTerms = GbifTerm.listByGroup(DwcTerm.GROUP_LOCATION);
53      assertEquals(6, locationTerms.size());
54      assertEquals(6, new HashSet<GbifTerm>(locationTerms).size());

56      List<GbifTerm> rowTerms = GbifTerm.listByGroup(GbifTerm.GROUP_ROW_TYPE);

81    @Test
82    public void testGroupCoverage() {
83      HashSet<GbifTerm> arrayTerms = new HashSet<GbifTerm>();
84      for (GbifTerm t : GbifTerm.TAXONOMIC_TERMS) {
85        arrayTerms.add(t);
86        assertFalse(t.isClass());
87        assertEquals(DwcTerm.GROUP_TAXON, t.getGroup());
88      }
89  
90      for (GbifTerm t : GbifTerm.listByGroup(DwcTerm.GROUP_TAXON)) {
91        if (!arrayTerms.contains(t)) {
92          assertTrue("Missing taxonomic term in GbifTerm.TAXONOMIC_TERMS: " + t.qualifiedName(), arrayTerms.contains(t));
93        }
94      }
95    }
