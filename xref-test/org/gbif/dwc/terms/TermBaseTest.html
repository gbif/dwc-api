3   import org.junit.Test;
4   
5   import static org.junit.Assert.assertEquals;
6   import static org.junit.Assert.assertFalse;

8   
9   /**
10   * Generic tests for all Term implementing classes.
11   */
12  public abstract class TermBaseTest<T extends Term> {
13  
14    private static final TermFactory TERM_FACTORY = TermFactory.instance();
15  
16    private final Class<T> clazz;
17    private final T[] values;
18    private final String[] forbiddenChars;
19    private final boolean skipSimple;
20  
21    public TermBaseTest(Class<T> clazz) {
22      this(clazz, new String[]{"_","-"}, false);
23    }
24  
25    public TermBaseTest(Class<T> clazz, String[] forbiddenChars, boolean skipSimple) {
26      this.clazz = clazz;
27      this.forbiddenChars = forbiddenChars;
28      values = clazz.getEnumConstants();
29      this.skipSimple = skipSimple;
30    }
31  
32    @Test
33    public void testNames() throws Exception {
34      for (T t : values) {
35        for (String c : forbiddenChars) {
36          assertFalse("Term contains forbidden character " + c + " : " + t.simpleName(), t.simpleName().contains(c));
37          assertFalse("Term contains forbidden character " + c + " : " + t.qualifiedName(), t.qualifiedName().contains(c));
38        }
39      }
40    }
41  
42    /**
43     * Test that iterates over all term values and uses the simple name alone as input to the factory and then verifies
44     * the Term returned is the same as the original one.
45     */
46    @Test
47    public void testFindSimpleTerm() throws Exception {
48      if (!skipSimple) {
49        for (T t : values) {
50            assertEquals(t, TERM_FACTORY.findTerm(t.simpleName(), t.isClass()));
51        }
52      }
53    }
54  
55    @Test
56    public void testFindPrefixedTerms() {
57      for (T t : values) {
58        Term found = TERM_FACTORY.findTerm(t.prefixedName(), t.isClass());
59        assertEquals(t, found);
60      }
61    }
62  
63    @Test
64    public void testFindQualifiedTerm() {
65      for (T t : values) {
66        assertEquals(t, TERM_FACTORY.findTerm(t.qualifiedName(), t.isClass()));
67      }
68    }
69  
70  }
