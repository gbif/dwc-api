1   package org.gbif.dwc.terms;

3   import org.junit.Test;
4   
5   import static org.junit.Assert.assertEquals;
6   import static org.junit.Assert.assertFalse;
7   
8   public class DcElementTest {
9   
10    private static final TermFactory TERM_FACTORY = TermFactory.instance();
11  
12    @Test
13    public void testNames() throws Exception {
14      for (DcElement t : DcElement.values()) {
15        assertFalse("Bad term: " + t.simpleName(), t.simpleName().contains("_"));
16        assertFalse("Bad term: " + t.qualifiedName(), t.qualifiedName().contains("_"));
17        assertFalse("Bad term: " + t.qualifiedName(), t.isClass());
18      }
19    }
20  
21    /**
22     * Test that iterates over all DcElement values and uses the qualified name alone as input to the factory and then
23     * verifies the Term returned is the same as the original one.
24     */
25    @Test
26    public void testTermEquality() throws Exception {
27      for (DcTerm t : DcTerm.values()) {
28        Term found = TERM_FACTORY.findTerm(t.qualifiedName());
29        assertEquals(t, found);
30      }
31    }
32  
33    /**
34     * A lookup for "http://purl.org/dc/elements/1.1/rights" shouldn't return "http://purl.org/dc/terms/rights"
35     */
36    @Test
37    public void testFindTermWithValidNamespace() {
38      String qualifiedName = "http://purl.org/dc/elements/1.1/rights";
39      Term term = TERM_FACTORY.findTerm(qualifiedName);
40      assertEquals(qualifiedName, term.qualifiedName());
41    }
42  
43    /**
44     * There are 15 terms in DcElement that share the same name with DcTerm elements. This test ensures, that
45     * when a term lookup is done with the prefix "dc", the DcElement under the /elements/1.1/ namespace
46     * is returned, not the DcTerm under the /terms namespace.
47     */
48    @Test
49    public void testFindTermWithDcPrefix() {
50      String name = "dc:rights";
51      Term term = TERM_FACTORY.findTerm(name);
52      assertEquals("http://purl.org/dc/elements/1.1/rights", term.qualifiedName());
53    }
54  }
